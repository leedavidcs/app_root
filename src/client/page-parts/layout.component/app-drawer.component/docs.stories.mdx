# AppDrawer

## Description
Side-drawer to be opened from the navbar, to be used for additional navigation.

## Props
<Props of={AppDrawer} />

## StandardStory
<Preview>
	<Story name="standard">
		{() => {
			const [isOpen, setIsOpen] = useState(false);
			const onClick = useCallback(() => setIsOpen(!isOpen), [isOpen, setIsOpen]);
			const onClose = useCallback(() => setIsOpen(false), [setIsOpen]);
			return (
				<>
					<AppDrawer isOpen={isOpen} onClose={onClose} />
					<Button intent="primary" onClick={onClick} text="Toggle app drawer" />
				</>
			);
		}}
	</Story>
</Preview>
